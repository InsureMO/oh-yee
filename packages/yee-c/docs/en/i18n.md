# I18n

Internationalization (I18n) utility class providing multi-language support and message formatting.

## Installation

```bash
npm install @oh/yee-tools
```

## Import

```typescript
import { I18nUtils } from '@oh/yee-tools/i18n';
```

## API

### getUserLang

Gets user language setting.

```typescript
static getUserLang(langUrl: string, isLogin?: boolean): string | null
```

**Example:**

```typescript
import { I18nUtils } from '@oh/yee-tools/i18n';

// Get user language
const lang = I18nUtils.getUserLang('/api/user/lang');
console.log(lang); // 'en_US' or 'zh_CN'

// Re-fetch after login
const langAfterLogin = I18nUtils.getUserLang('/api/user/lang', true);
```

---

### getI18nData

Gets internationalization data.

```typescript
static getI18nData(lang: string, i18nUrl: string): I18nData
```

**Example:**

```typescript
import { I18nUtils } from '@oh/yee-tools/i18n';

// Get i18n data
const i18nData = I18nUtils.getI18nData('en_US', '/api/i18n');

// Use translations
console.log(i18nData.welcome); // 'Welcome'
console.log(i18nData.goodbye); // 'Goodbye'
```

### format

Formats message with placeholder replacement.

```typescript
static format(message: string, ...args: any[]): string
```

**Example:**

```typescript
import { I18nUtils } from '@oh/yee-tools/i18n';

// Simple replacement
const msg1 = I18nUtils.format('Hello {0}', 'John');
// => 'Hello John'

// Multiple parameters
const msg2 = I18nUtils.format('Hello {0}, you have {1} messages', 'John', 5);
// => 'Hello John, you have 5 messages'

// With i18n data
const i18nData = { welcome: 'Welcome {0} to {1}' };
const msg3 = I18nUtils.format(i18nData.welcome, 'John', 'YeeTools');
// => 'Welcome John to YeeTools'
```

---

### formatObject

Formats object message, returns array with text and objects.

```typescript
static formatObject(message: string, ...args: any[]): any[]
```

**Example:**

```typescript
import { I18nUtils } from '@oh/yee-tools/i18n';

const button = { type: 'button', text: 'Click me' };
const elements = I18nUtils.formatObject('Click {0} to continue', button);
// => ['Click ', button, ' to continue']
```

---

### getMessage

Gets internationalization message, handles null values.

```typescript
static getMessage(message: string | null | undefined): string
```

**Example:**

```typescript
import { I18nUtils } from '@oh/yee-tools/i18n';

const msg1 = I18nUtils.getMessage(null);
// => 'MSG Not Found'

const msg2 = I18nUtils.getMessage('Hello');
// => 'Hello'
```

## Common Use Cases

### Initialize I18n

```typescript
import { I18nUtils, configer } from '@oh/yee-tools';

// Configure
configer.setConfig({
  i18n: {
    defaultLocale: 'en_US',
    storageKey: 'system_i18nKey',
    configurationGroup: ['common', 'business'],
  },
});

// Initialize
function initI18n() {
  const lang = I18nUtils.getUserLang('/api/user/lang');
  const i18nData = I18nUtils.getI18nData(lang, '/api/i18n');

  // Store to cache
  SessionContext.put('i18nData', i18nData);
  SessionContext.put('currentLang', lang);
}
```

### Change Language

```typescript
import { I18nUtils } from '@oh/yee-tools/i18n';

async function changeLanguage(newLang: string) {
  // Update system language
  I18nUtils.setSystemI18N(newLang);

  // Get new i18n data
  const i18nData = I18nUtils.getI18nData(newLang, '/api/i18n');

  // Update app state
  SessionContext.put('i18nData', i18nData);

  // Refresh page
  window.location.reload();
}
```

### Translation Function

```typescript
import { I18nUtils } from '@oh/yee-tools/i18n';

function t(key: string, ...args: any[]): string {
  const i18nData = SessionContext.get('i18nData');
  const message = i18nData?.[key] || key;

  if (args.length > 0) {
    return I18nUtils.format(message, ...args);
  }

  return message;
}

// Usage
const welcome = t('welcome'); // 'Welcome'
const greeting = t('greeting', 'John'); // 'Hello John'
```

### Dynamic Message Formatting

```typescript
import { I18nUtils } from '@oh/yee-tools/i18n';

// User welcome message
function getUserMessage(username: string, messageCount: number): string {
  const template = 'Hello {0}, you have {1} new messages';
  return I18nUtils.format(template, username, messageCount);
}

// Form validation message
function getRequiredError(fieldName: string): string {
  return I18nUtils.format('{0} is required', fieldName);
}
```

### React Integration Example

```typescript
import { I18nUtils } from '@oh/yee-tools/i18n';
import { useState, useEffect } from 'react';

function useI18n() {
  const [lang, setLang] = useState(I18nUtils.getSystemI18N());
  const [messages, setMessages] = useState({});

  useEffect(() => {
    const i18nData = I18nUtils.getI18nData(lang, '/api/i18n');
    setMessages(i18nData);
  }, [lang]);

  const t = (key: string, ...args: any[]) => {
    const message = messages[key] || key;
    return I18nUtils.format(message, ...args);
  };

  const changeLang = (newLang: string) => {
    I18nUtils.setSystemI18N(newLang);
    setLang(newLang);
  };

  return { lang, t, changeLang };
}

// Usage
function App() {
  const { t, changeLang } = useI18n();

  return (
    <div>
      <h1>{t('welcome')}</h1>
      <button onClick={() => changeLang('en_US')}>English</button>
      <button onClick={() => changeLang('zh_CN')}>中文</button>
    </div>
  );
}
```

## Configuration Options

```typescript
interface I18nConfig {
  defaultLocale?: string; // Default language 'en_US'
  storageKey?: string; // Storage key 'system_i18nKey'
  configurationGroup?: string[]; // Configuration group ['common']
  cache?: boolean; // Enable cache true
}
```

## Supported Language Codes

- `en_US` - English (United States)
- `zh_CN` - Simplified Chinese
- `ja_JP` - Japanese
- `ko_KR` - Korean
- Other custom language codes

## Notes

1. **Synchronous requests**: `getUserLang` and `getI18nData` use synchronous requests by default
2. **Caching**: I18n data is automatically cached to LocalContext and PageContext
3. **Error handling**: 403/401 responses trigger logout logic
4. **Configuration dependency**: Requires `configer` i18n options to be configured first
